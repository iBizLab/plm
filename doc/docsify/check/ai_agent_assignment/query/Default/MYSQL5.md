```sql
SELECT
t11.`CODE_NAME` AS `CONTEXT_CODE_NAME`,
t1.`CONTEXT_ID`,
t11.`NAME` AS `CONTEXT_NAME`,
t1.`CREATE_MAN`,
t1.`CREATE_TIME`,
t11.`DESCRIPTION`,
t1.`ID`,
t1.`NAME`,
t1.`SYSTEM_FLAG`,
t1.`UPDATE_MAN`,
t1.`UPDATE_TIME`,
t1.`USE_TAG`
FROM `AI_AGENT_ASSIGNMENT` t1 
LEFT JOIN `AI_AGENT_CONTEXT` t11 ON t1.`CONTEXT_ID` = t11.`ID`
```
